1.1 Cleaning and variable selection

from pyspark.sql.functions import when, col, pow, lower, trim
from pyspark.ml.feature import StringIndexer, OneHotEncoder, VectorAssembler
from pyspark.ml import Pipeline
from pyspark.ml.regression import LinearRegression
from pyspark.ml.evaluation import RegressionEvaluator

cont_vars = ["MIN_YEARS_EXPERIENCE", "MAX_YEARS_EXPERIENCE", "DURATION", "MIN_EDULEVELS"]
cat_vars = ["EMPLOYMENT_TYPE_NAME", "REMOTE_TYPE_NAME"]
y = ["SALARY"]

median_duration = df.approxQuantile("DURATION", [0.5], 0.01)[0]

df = df.withColumn(
    "DURATION",
    when(col("DURATION").isNull(), median_duration)
    .otherwise(col("DURATION"))
)

median_salary = df.approxQuantile("SALARY", [0.5], 0.01)[0]

df = df.withColumn(
    "SALARY",
    when(
        col("SALARY").isNull() &
        (col("SALARY_TO").isNull() | col("SALARY_FROM").isNull()),
        median_salary
    )
    .when(col("SALARY").isNull() & col("SALARY_TO").isNull(), col("SALARY_FROM"))
    .when(col("SALARY").isNull() & col("SALARY_FROM").isNull(), col("SALARY_TO"))
    .when(
        col("SALARY").isNull() &
        col("SALARY_TO").isNotNull() &
        col("SALARY_FROM").isNotNull(),
        (col("SALARY_TO") + col("SALARY_FROM")) / 2
    )
    .otherwise(col("SALARY"))
)

df = df.withColumn(
    "REMOTE_TYPE_NAME",
    when(
        col("REMOTE_TYPE_NAME").isNull() | (trim(col("REMOTE_TYPE_NAME")) == "") | (trim(lower(col("REMOTE_TYPE_NAME"))).rlike(r"\[?none\]?")),
        "Undefined"
    )
    .when(lower(col("REMOTE_TYPE_NAME")).contains("not remote"), "Onsite")
    .when(lower(col("REMOTE_TYPE_NAME")).contains("hybrid"), "Hybrid")
    .otherwise(col("REMOTE_TYPE_NAME"))
)

df = df.withColumn(
    "EMPLOYMENT_TYPE_NAME",
    when(col("EMPLOYMENT_TYPE_NAME") == "Part-time / full-time", "Flexible")
    .when(col("EMPLOYMENT_TYPE_NAME") == "Part-time (â‰¤ 32 hours)", "Part-time")
    .when(col("EMPLOYMENT_TYPE_NAME") == "Full-time (> 32 hours)", "Full-time")
    .when(col("EMPLOYMENT_TYPE_NAME").isNull(), "Fulltime")
    .otherwise(col("EMPLOYMENT_TYPE_NAME")
))

df = df.withColumn("MIN_YEARS_EXPERIENCE_SQ", pow(col("MIN_YEARS_EXPERIENCE"), 2))

df = df.withColumn(
    "MAX_YEARS_EXPERIENCE",
    when(
        col("MAX_YEARS_EXPERIENCE").isNull() & col("MIN_YEARS_EXPERIENCE").isNotNull(),
        col("MIN_YEARS_EXPERIENCE") + 1 
    ).otherwise(col("MAX_YEARS_EXPERIENCE"))
)

df = df.dropna(subset = cont_vars + cat_vars + y).select(cont_vars + cat_vars + y)

Four continuous variables were selected based on their relevance to compensation. Null entries were imputed using median values and logical rules in order to maximize the number of valid entries to include in the training set. Categorical variables were encoded using StringIndexer and OneHotEncoder, converting them into numeric vectors suitable for model input.

Interpretation

Comparison of these values reveals that, between the three models, the Random Forest model achieved the best performance, reducing RMSE by nearly 5% compared to polynomial regression.

Random Forest also had the lowest AIC and BIC values which indicates a good trade-off between model complexity and fit.

Linear and polynomial models revealed interpretable relationships, but the data’s interactions were more complex than these models allowed, resulting in only ~12% of explainable variance in salary. Random Forest was able to capture these patterns more effectively.

7.1 Calculating log-likelihood and BIC

from math import log

def calc_log_likelihood(y_true, y_pred, sigma2):
    n = len(y_true)
    ll = -0.5 * n * log(2 * np.pi * sigma2) - 0.5 * np.sum((y_true - y_pred) ** 2) / sigma2
    return ll

def get_y_true_pred(results_df, pred_col):
    pdf = results_df.select("actual", pred_col).toPandas()
    y_true = pdf["actual"].values
    y_pred = pdf[pred_col].values
    return y_true, y_pred

y_true_glr, y_pred_glr = get_y_true_pred(glr_results, "pred_glr")
sigma2_glr = np.var(y_true_glr - y_pred_glr)
ll_glr = calc_log_likelihood(y_true_glr, y_pred_glr, sigma2_glr)
aic_glr = 2 * len(glr_model.coefficients) - 2 * ll_glr
bic_glr = log(len(y_true_glr)) * len(glr_model.coefficients) - 2 * ll_glr

y_true_poly, y_pred_poly = get_y_true_pred(poly_results, "pred_poly")
sigma2_poly = np.var(y_true_poly - y_pred_poly)
ll_poly = calc_log_likelihood(y_true_poly, y_pred_poly, sigma2_poly)
aic_poly = 2 * len(poly_model.coefficients) - 2 * ll_poly
bic_poly = log(len(y_true_poly)) * len(poly_model.coefficients) - 2 * ll_poly

y_true_rf, y_pred_rf = get_y_true_pred(rf_results, "pred_rf")
sigma2_rf = np.var(y_true_rf - y_pred_rf)
ll_rf = calc_log_likelihood(y_true_rf, y_pred_rf, sigma2_rf)
n_rf_params = len(rf_model.featureImportances)
aic_rf = 2 * n_rf_params - 2 * ll_rf
bic_rf = log(len(y_true_rf)) * n_rf_params - 2 * ll_rf
